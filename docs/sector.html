    // --- 2. FUNCIONES DE FORMATO ---
    const fmtNum = (n) => new Intl.NumberFormat('es-ES', { maximumFractionDigits: 0 }).format(n);
    const fmtEur = (n) => new Intl.NumberFormat('es-ES', { style: 'currency', currency: 'EUR', maximumFractionDigits: 0 }).format(n);
    const calcCAGR = (start, end, periods) => (start > 0 && periods > 0) ? ((Math.pow(end / start, 1 / periods) - 1) * 100).toFixed(2) : 0;

    // --- 3. INICIALIZACIÓN Y NAVEGACIÓN ---
    function switchTab(tabId) {
        if(tabId === 'data') return; // Bloquear pestaña de datos
        ['dashboard', 'comparator'].forEach(t => {
            document.getElementById('view-' + t).classList.add('hidden');
            document.getElementById('btn-' + t).className = 'tab-inactive px-4 py-2 rounded text-sm font-bold transition-all flex items-center gap-2 cursor-pointer';
        });
        document.getElementById('view-' + tabId).classList.remove('hidden');
        document.getElementById('btn-' + tabId).className = 'tab-active px-4 py-2 rounded text-sm font-bold transition-all flex items-center gap-2 cursor-default';
        
        if(tabId === 'comparator') updateComparatorChart();
    }

    // --- 4. LÓGICA DASHBOARD PRINCIPAL ---
    function renderDashboard() {
        let totalSales2024 = 0;
        let totalSales2023 = 0;
        let topGrower = { name: '-', growth: -Infinity };

        DB.forEach(c => {
            const s24 = c.sales[2024] || 0;
            const s23 = c.sales[2023] || 0;
            totalSales2024 += s24;
            totalSales2023 += s23;
            
            const growth = s24 - s23;
            if (growth > topGrower.growth) topGrower = { name: c.name, growth: growth };
        });

        document.getElementById('kpi-total-companies').innerText = DB.length;
        document.getElementById('kpi-total-sales').innerText = fmtNum(totalSales2024) + ' GWh';
        document.getElementById('kpi-total-cost').innerText = fmtEur(totalSales2024 * 239000).slice(0,-2) + ' M€';
        document.getElementById('kpi-top-grower-name').innerText = topGrower.name;
        document.getElementById('kpi-top-grower-val').innerText = '+' + fmtNum(topGrower.growth) + ' GWh';

        renderTable(DB);
        renderMarketChart();
    }

    function renderTable(data) {
        const tbody = document.getElementById('companies-table-body');
        tbody.innerHTML = '';
        const sorted = [...data].sort((a,b) => (b.sales[2024]||0) - (a.sales[2024]||0)).slice(0, 200);

        sorted.forEach(c => {
            const s24 = c.sales[2024] || 0;
            const s23 = c.sales[2023] || 0;
            const trend = s23 > 0 ? ((s24 - s23)/s23 * 100).toFixed(1) : 0;
            const trendColor = trend >= 0 ? 'text-emerald-600' : 'text-rose-600';
            
            const row = `
                <tr class="hover:bg-blue-50 transition-colors border-b border-slate-100">
                    <td class="px-6 py-3 font-mono text-xs text-slate-500">${c.cif}</td>
                    <td class="px-6 py-3 font-medium text-slate-700 truncate max-w-xs" title="${c.name}">${c.name}</td>
                    <td class="px-6 py-3 text-right font-bold text-slate-800">${fmtNum(s24)}</td>
                    <td class="px-6 py-3 text-right font-bold text-xs ${trendColor}">${trend > 0 ? '+' : ''}${trend}%</td>
                    <td class="px-6 py-3 text-center">
                        <button onclick="openModal('${c.cif}')" class="text-blue-600 hover:bg-blue-100 p-2 rounded transition">
                            <i class="fa-solid fa-eye"></i>
                        </button>
                    </td>
                </tr>
            `;
            tbody.innerHTML += row;
        });
    }

    function renderMarketChart() {
        const ctx = document.getElementById('marketChart').getContext('2d');
        const totals = YEARS.map(y => DB.reduce((sum, c) => sum + (c.sales[y]||0), 0));

        if(marketChartInstance) marketChartInstance.destroy();

        // Chart en modo oscuro
        marketChartInstance = new Chart(ctx, {

    function filterTable() {
        const term = document.getElementById('search-input').value.toLowerCase();
        const filtered = DB.filter(c => c.name.toLowerCase().includes(term) || c.cif.toLowerCase().includes(term));
        renderTable(filtered);
    }

    // --- 5. LÓGICA COMPARADOR ---
    function searchCompetitor() {
        const term = document.getElementById('comp-search').value.toLowerCase();
        const resultsDiv = document.getElementById('comp-results');
        resultsDiv.innerHTML = '';
        
        if(term.length < 2) { resultsDiv.classList.add('hidden'); return; }

        const matches = DB.filter(c => c.name.toLowerCase().includes(term) && !selectedCompetitors.includes(c.cif)).slice(0, 10);
        
        if(matches.length > 0) {
            resultsDiv.classList.remove('hidden');
            // Fondo oscuro para resultados
            resultsDiv.classList.add('bg-slate-800', 'border-slate-700'); 
            matches.forEach(c => {
                const div = document.createElement('div');
                div.className = 'p-3 hover:bg-slate-700 cursor-pointer text-xs border-b border-slate-700 text-slate-200';
                div.innerHTML = `<strong>${c.name}</strong> <span class="text-slate-400">(${c.cif})</span>`;
                div.onclick = () => addCompetitor(c.cif);
                resultsDiv.appendChild(div);
            });
        } else {
            resultsDiv.classList.add('hidden');
        }
    }

    function addCompetitor(cif) {
        if(selectedCompetitors.length >= 5) { alert("Máximo 5 competidores"); return; }
        selectedCompetitors.push(cif);
        document.getElementById('comp-search').value = '';
        document.getElementById('comp-results').classList.add('hidden');
        renderSelectedChips();
        updateComparatorChart();
    }

    function removeCompetitor(cif) {
        selectedCompetitors = selectedCompetitors.filter(c => c !== cif);
        renderSelectedChips();
        updateComparatorChart();
    }

    function renderSelectedChips() {
        const container = document.getElementById('selected-comps');
        container.innerHTML = '';
        const colors = ['#2dd4bf', '#f472b6', '#34d399', '#fbbf24', '#a78bfa']; // Paleta Neon
        
        selectedCompetitors.forEach((cif, idx) => {
            const comp = DB.find(c => c.cif === cif);
            const div = document.createElement('div');
            div.className = 'flex items-center justify-between p-2 bg-slate-800 rounded border border-slate-700 text-xs text-slate-200';
            div.innerHTML = `
                <div class="flex items-center gap-2 overflow-hidden">
                    <div class="w-3 h-3 rounded-full flex-shrink-0" style="background:${colors[idx]}"></div>
                    <span class="font-bold truncate">${comp.name}</span>
                </div>
                <button onclick="removeCompetitor('${cif}')" class="text-slate-400 hover:text-red-500"><i class="fa-solid fa-times"></i></button>
            `;
            container.appendChild(div);
        });
    }

    function setChartMode(mode) {
        comparatorMode = mode;
        document.getElementById('mode-abs').className = mode === 'absolute' ? 'w-full text-left px-4 py-3 rounded mb-2 text-sm font-medium transition-colors tab-active' : 'w-full text-left px-4 py-3 rounded mb-2 text-sm font-medium transition-colors tab-inactive';

    }

    function updateComparatorChart() {
        const ctx = document.getElementById('comparatorChart').getContext('2d');
        const colors = ['#2dd4bf', '#f472b6', '#34d399', '#fbbf24', '#a78bfa'];
        
        const datasets = selectedCompetitors.map((cif, idx) => {
            const comp = DB.find(c => c.cif === cif);
            let data = YEARS.map(y => comp.sales[y] || 0);
            
            if (comparatorMode === 'indexed') {
                const base = data[0] || 1; 
                data = data.map(v => (v/base)*100);
            }

            return {
                label: comp.name,
                data: data,
                borderColor: colors[idx],
                backgroundColor: colors[idx],
                borderWidth: 2,
                tension: 0.3,
                pointRadius: 4
            };
        });

        if(comparatorChartInstance) comparatorChartInstance.destroy();

        comparatorChartInstance = new Chart(ctx, {
            type: 'line',
            data: { labels: YEARS, datasets: datasets },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                interaction: { mode: 'index', intersect: false },
                scales: { 
                    y: { 
                        beginAtZero: false, 
                        grid: { color: '#334155' },
                        ticks: { color: '#94a3b8' }
                    }, 
                    x: { 
                        grid: { display: false },
                        ticks: { color: '#94a3b8' }
                    } 
                },
                plugins: { legend: { labels: { color: '#e2e8f0' } } }
            }
        });

        // Actualizar tabla comparativa (Estilo oscuro)
        const tbody = document.getElementById('comp-table-body');
        tbody.innerHTML = '';
        selectedCompetitors.forEach(cif => {
            const comp = DB.find(c => c.cif === cif);
            const s20 = comp.sales[2020] || 0;
            const s24 = comp.sales[2024] || 0;
            const growth = s20 > 0 ? ((s24 - s20)/s20 * 100).toFixed(1) : 0;
            const cagr = calcCAGR(s20, s24, 4);
            
            const row = `
                <tr class="border-b border-slate-700">
                    <td class="px-4 py-2 font-bold text-slate-200 truncate max-w-xs">${comp.name}</td>
                    <td class="px-4 py-2 text-right text-slate-400">${fmtNum(s20)}</td>
                    <td class="px-4 py-2 text-right font-bold text-slate-200">${fmtNum(s24)}</td>
                    <td class="px-4 py-2 text-right text-emerald-400 font-bold">${cagr}%</td>
                    <td class="px-4 py-2 text-right ${growth >= 0 ? 'text-emerald-400' : 'text-rose-400'} font-bold">${growth}%</td>
                </tr>
            `;
            tbody.innerHTML += row;
        });
    }

    // --- 6. MODAL DETALLE ---
    function openModal(cif) {
        const comp = DB.find(c => c.cif === cif);
        if(!comp) return;

        document.getElementById('modal').classList.remove('hidden');
        document.getElementById('modal-cif').innerText = comp.cif;
        document.getElementById('modal-name').innerText = comp.name;
        
        const s20 = comp.sales[2020] || 0;
        const s24 = comp.sales[2024] || 0;
        const growth = s20 > 0 ? ((s24 - s20)/s20 * 100).toFixed(1) + '%' : '-';
        const cagr = calcCAGR(s20, s24, 4) + '%';
        
        document.getElementById('modal-sales-24').innerText = fmtNum(s24);
        document.getElementById('modal-growth').innerText = growth;
        document.getElementById('modal-cagr').innerText = cagr;
        document.getElementById('modal-cost').innerText = fmtEur(s24 * 239000);

        const ctx = document.getElementById('modalChart').getContext('2d');
        if(modalChartInstance) modalChartInstance.destroy();

        modalChartInstance = new Chart(ctx, {
            type: 'line',
            data: {
                labels: YEARS,
                datasets: [{
                    label: 'Ventas (GWh)',
                    data: YEARS.map(y => comp.sales[y]||0),
                    borderColor: '#2dd4bf',
                    backgroundColor: 'rgba(45, 212, 191, 0.1)',
                    fill: true,
                    tension: 0.4
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: { legend: { display: false } },
                scales: { 
                    y: { beginAtZero: true, grid: { color: '#334155' }, ticks: { color: '#94a3b8' } },
                    x: { grid: { display: false }, ticks: { color: '#94a3b8' } }
                }
            }
        });
    }

    function closeModal() {
        document.getElementById('modal').classList.add('hidden');
    }
