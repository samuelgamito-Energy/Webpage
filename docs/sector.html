// 2. LIMPIEZA DE FORMATO (L√≠nea a l√≠nea)
clean_csv = raw.split(/\r?\n/).map(line => {
  let l = line.trim();
  if (!l) return null; 
  
  // Si la l√≠nea est√° entre comillas dobles, las quitamos
  if (l.startsWith('"') && l.endsWith('"')) {
    l = l.slice(1, -1);
  }
  
  // Si sobra un punto y coma al final, lo quitamos
  if (l.endsWith(';')) {
    l = l.slice(0, -1);
  }
  
  return l;
}).filter(l => l !== null).join("\n")

// 3. PARSEO Y CONVERSI√ìN NUM√âRICA
data = d3.dsvFormat(";").parse(clean_csv, (d) => {
  for (const key in d) {
    // Detectamos columnas de Ventas
    if (key.includes("Ventas") && d[key]) {
      // Formato espa√±ol: 1.000,00 -> 1000.00
      let val = d[key].toString()
                     .replace(/\./g, "")   // Quitar puntos de miles
                     .replace(",", ".");   // Cambiar coma por punto decimal
      d[key] = parseFloat(val) || 0;
    }
  }
  return d;
})

// 4. DATOS FORMATO LARGO (Para gr√°ficas)
