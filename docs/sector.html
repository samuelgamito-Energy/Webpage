data = d3.dsvFormat(";").parse(csvData, (d) => {
  const row = { "NIF": d.NIF, "Empresa": d.Empresa };
  Object.keys(d).forEach(k => {
    if (k.includes("Ventas")) {
      let year = k.replace("Ventas", "").trim();
      let val = parseFloat(d[k].replace(",", ".")) || 0;
      row[year] = val;
    }
  });
  return row;
})
