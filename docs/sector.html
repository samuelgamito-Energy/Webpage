raw = await FileAttachment("ventas_boe.csv").text()

// Esta funci√≥n limpia el CSV l√≠nea por l√≠nea antes de que D3 intente leerlo
clean_csv = raw.split(/\r?\n/).map(line => {
  let l = line.trim();
  if (!l) return null; // Ignorar l√≠neas vac√≠as
  
  // 1. Si la l√≠nea entera tiene comillas al principio y final, las quitamos
  // Ejemplo: "dato;dato;" -> dato;dato;
  if (l.startsWith('"') && l.endsWith('"')) {
    l = l.slice(1, -1);
  }
  
  // 2. Si hay un punto y coma al final (muy com√∫n en Excel), lo quitamos
  if (l.endsWith(';')) {
    l = l.slice(0, -1);
  }
  
  return l;
}).filter(l => l !== null).join("\n")

// --- 2. PARSEO DE DATOS ---
data = d3.dsvFormat(";").parse(clean_csv, (d, i, columns) => {
  // Forzamos columnas por posici√≥n para evitar errores de nombres
  const row = {
    "NIF": d[columns[0]],       // Columna 0 es NIF
    "Empresa": d[columns[1]]    // Columna 1 es Nombre
  };

  // Buscamos columnas de ventas
  columns.forEach(col => {
    if (col.includes("Ventas")) {
      // Limpiamos el n√∫mero europeo: 1.234,56 -> 1234.56
      let valStr = d[col] ? d[col].toString() : "0";
      let val = parseFloat(valStr.replace(/\./g, "").replace(",", ".")) || 0;
      
      // Sacamos el a√±o del nombre (ej: "Ventas 2024" -> "2024")
      let year = col.replace("Ventas", "").trim();
      row[year] = val; 
    }
  });
  return row;
})

// --- 3. FILTRO DE A√ëOS (Borramos 2025) ---
